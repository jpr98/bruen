function fib(n: int) int {
    if (n == 1 || n == 0) {
        return n;
    }
    return fib(n - 1) + fib(n - 2);
}

function factorial(n: int) int {
    if (n < 2) {
        return 1;
    }
    return n * factorial(n - 1);
}
